a) 	SELECT p.product_name, p.product_id, c.category_name, p.unit_price 
		FROM products p 
		JOIN categories c 
		ON p.category_id = c.category_id 
		ORDER BY p.product_name;

		FROM products p 
		JOIN categories c 
		ON p.category_id = c.category_id 
		WHERE LOWER(c.category_name) = 'dairy products' AND p.unit_price < 30 
		ORDER BY p.unit_price;

Exercice 3.2:
a) 	SELECT last_name, first_name 
		FROM employees 
		WHERE employee_id IN (
			SELECT employee_id 
				FROM employee_territories et 
				JOIN territories t 
				ON et.territory_id = t.territory_id 
				JOIN region r 
				ON t.region_id = r.region_id 
				WHERE LOWER(r.region_description) = 'northern'
		);

b) 	Ici on utilise un `LEFT JOIN` pour inclure les catégories sans produits.
	SELECT c.category_name, COUNT(*) AS "products_count" 
		FROM categories c 
		LEFT JOIN products p ON c.category_id = p.category_id 
		GROUP BY c.category_id 
		ORDER BY products_count DESC;

c) 	SELECT p.product_name, SUM(od.quantity) 
		FROM products p 
		JOIN order_details od 
		ON p.product_id = od.product_id 
		GROUP BY p.product_id 
		ORDER BY SUM(od.quantity) DESC 
		LIMIT 10;

bonus) 	SELECT p.product_name, SUM(od.quantity) 
		FROM products p 
		JOIN order_details od 
		ON p.product_id = od.product_id 
		JOIN suppliers s 
		ON p.supplier_id = s.supplier_id 
		WHERE LOWER(s.country) = 'france' 
		GROUP BY p.product_id 
		ORDER BY SUM(od.quantity) DESC 
		LIMIT 5;
