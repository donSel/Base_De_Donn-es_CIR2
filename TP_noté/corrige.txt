 - 3.2

a) select e.first_name, e.last_name 
   from employees e 
   join employee_territories et 
   on e.employee_id=et.employee_id 
   join territories t
   on et.territory_id=t.territory_id 
   where t.region_id in (select r.region_id from region r where lower(r.region_description)='northern');
b) SELECT c.category_name, COUNT(*) AS "products_count"
   FROM categories c
   LEFT JOIN products p ON c.category_id = p.category_id
   GROUP BY c.category_id
   ORDER BY products_count DESC;
c) select top p.product_name, sum(o.quantity) as number_sells from order_details o natural join products p group by p.product_id order by number_sells desc limit 10*;
