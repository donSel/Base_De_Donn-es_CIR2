 -- ex3

a)
alter table employe add salaireetcom numeric(7,2);

b)
create function wrong_salaire() return trigger as $wrong_salaire$
begin
if new.salaireetcom < new.salire then
raise exception '% ne peut pas avoir un salaire + commission inferieur au salaire !', new.nom;
end if;
return new;
end;
$wrong_salaire$ language plpsql;

create trigger wrong_salaire before insert or update on employe for each row execute procedure wrong_salaire();
