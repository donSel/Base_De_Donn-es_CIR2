a)
SELECT num_service, COUNT(*)
FROM employe
GROUP BY num_service;

b)
SELECT s.ville, COUNT(*), ROUND(AVG(e.salaire), 2) 
FROM service s, employe e
WHERE e.num_service = s.num_service
GROUP BY s.ville
ORDER BY s.ville;

a)
SELECT num_service, type_service
FROM service
WHERE num_service NOT IN (SELECT num_service
			  FROM employe);

a)
SELECT e.nom, e.fonction, es.nom as "nom_superieur", es.fonction as "fonction_superieur", s.ville
FROM employe e LEFT JOIN employe es
ON e.numemp_sup = es.numemp 
JOIN service s ON e.num_service = s.num_service
ORDER BY e.fonction, e.nom;
